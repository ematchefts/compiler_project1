int process (int n, int m)
{ 
  if (n+1 <= 0) return void ;
  else{
	if(n != m){
		n = n * m;
	}
	if(n >= m){
		n = n / m;
	}
	while(n < n*m){
		n = n+n;
	}
	while(n > m){
		n = n - 1;
	}
	if(n == 1){
		n = n;
	}
  } 
  return process(n-1);
}
